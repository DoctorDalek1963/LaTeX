\begin{center}
For integrals of the form on the left, consider the function on the right.
\end{center}
\vspace{-1em}

{\large\renewcommand{\arraystretch}{1.8}
$$\begin{array}{|c|c|}
	\hline
	\inte{k f' f^n}{x} & f^{n + 1} \\
	\hline
	\inte{k \frac{f'}{f}}{x} & \ln|f| \\
	\hline
\end{array}$$}
