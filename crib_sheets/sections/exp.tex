\begin{figure}[h]

	\begin{tikzpicture}
		\begin{axis}[

		\node at (yex title) [below right] {$y = x$};
	\end{tikzpicture}
\end{minipage}\hfill
\begin{minipage}{0.39\linewidth}
\large\centering
$$\diff{x} e^x = e^x$$

$$\diff{x} e^{kx} = k e^{kx}$$

$$\diff{x} e^{f(x)} = f'(x) e^{f(x)}$$

\vspace{1em}
$\ln x$ is the inverse of $e^x$, meaning its graph is reflected in the line $y = x$.
\end{minipage}
\end{figure}

\sectitle{Log Laws}

{\large $$\log_a b \equiv \frac{\ln b}{\ln a}$$}

\begin{figure}[h]
{\centering\large
\begin{minipage}{0.33\linewidth}
	$$\log xy \equiv \log x + \log y$$
	\vspace{0.5em}
	$$\log_a a \equiv 1$$
\end{minipage}\hfill
\begin{minipage}{0.33\linewidth}
	$$\log \frac{x}{y} \equiv \log x - \log y$$
	\vspace{0.5em}
	$$\log 1 \equiv 0$$
\end{minipage}\hfill
\begin{minipage}{0.33\linewidth}
	$$\log x^y \equiv y \log x$$
	\vspace{0.5em}
	$$\log \frac{1}{x} \equiv -\log x$$
\end{minipage}
}
\end{figure}
