		\inte{\arctan x}{x} &= x \arctan x - \frac{\ln(x^2 + 1)}{2} + C

	\diff{x}\sec x &= \sec x\tan x\ \ \ &
	\inte{\sec x}{x} &= \ln|\tan x + \sec x| + C\ \ \ &

	\diff{x}\arctan x &= \frac{1}{x^2 + 1}\\
