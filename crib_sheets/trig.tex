\usepackage{tikz}

\usepackage{pgfplots}
\pgfplotsset{
	compat=1.18,
	sincosaxis/.style={
		% Meant to be used in minipage, so full width
		width=\textwidth,
		height=0.8\textwidth,
		grid=both,
		samples=1000,
		no marks,
		ymax=1.2,
		ymin=-1.2,
		xmax=6.48,
		xmin=-6.48,
		xlabel=$x$,
		ylabel=$y$,
		xtick={-2*\npi,-1.5*\npi,...,2*\npi},
		xticklabels={%
			$-2\pi$,
			$-\frac{3}{2}\pi$,
			$-\pi$,
			$-\frac{1}{2}\pi$,
			0,
			$\frac{1}{2}\pi$,
			$\pi$,
			$\frac{3}{2}\pi$,
			$2\pi$},
		ytick={%
			-1,
			-0.5,
			0,
			0.5,
			1},
		yticklabels={%
			-1,
			$-\frac{1}{2}$,
			0,
			$\frac{1}{2}$,
			1},
		axis lines=middle
	},
	sincosplot/.style={
		domain=-2*\npi:2*\npi,
		thick
	}
}
