\documentclass[border=0.2cm]{standalone}

\usepackage{tikz}
\usetikzlibrary{math}
\tikzmath{
	\sf = sqrt(2)/2;
	\angle = 10;
}

\begin{document}

\centering
\begin{tikzpicture}
	\draw [ultra thin] (-1,-1) rectangle (1,1);
	\fill [gray!60] (-1,-1) rectangle (1,1);
	\fill [white] (0,0) circle[radius=1];

	\foreach \iter in {1,2,...,30} {
		\draw [rotate={\angle*\iter},ultra thin] ({-1*\sf^\iter},{-1*\sf^\iter}) rectangle ({\sf^\iter},{\sf^\iter});
		\fill [rotate={\angle*\iter},gray!60] ({-1*\sf^\iter},{-1*\sf^\iter}) rectangle ({\sf^\iter},{\sf^\iter});
		\fill [white] (0,0) circle[radius={\sf^\iter}];
	}
\end{tikzpicture}

\end{document}
